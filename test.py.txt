import os
from langchain_mistralai import ChatMistralAI

key = os.environ.get("MISTRAL_API_KEY")
print("Key seen by Python:", (key[:8] + "...") if key else "NONE")

llm = ChatMistralAI(model="mistral-medium-latest", mistral_api_key=key)
resp = llm.invoke("say hi")
print("Mistral replied:", resp.content)
